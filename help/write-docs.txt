desc:
  genarate basic sh file
args:
  --project-dir optional,set the project name
  -d,--debug optional,set the debug mode
  -h,--help optional,get the cmd help
how-to-run:
  run as shell args
    bash ./write-docs.sh
  run as runable application
    ./write-docs.sh --project-dir eth0
demo-with-args:
  without-args
    ok:./write-docs.sh 
  passed arg with necessary value
    ok:./write-docs.sh --project-dir eth0
    ok:./write-docs.sh --project-dir=eth0
  passed arg with optional value
  passed arg without value
how-to-get-help:
  ok:./write-docs.sh --help
  ok:./write-docs.sh -h
  ok:./write-docs.sh --debug
